* typeof 获取变量数据类型

  ```
  	var n = 10;
      var s = "string";
      var bool = true;
      var nul = null;
      var und;
      var obj = new Object();
      console.log(typeof n); // number
      console.log(typeof s); //  string
      console.log(typeof bool); //  boolean
      console.log(typeof nul); //  object
      console.log(typeof und); //  undefined
      console.log(typeof obj); //  object
  ```
